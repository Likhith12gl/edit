<script>
    $(document).ready(function () {
        // Handling Add Project Form Submission
        $('#addNewProjectForm').submit(function (event) {
            event.preventDefault();
            var projectName = $('#projectName').val();
            var projectDeadline = $('#projectDeadline').val();
            $.ajax({
                url: '/project/save',
                type: 'POST',
                data: { name: projectName, deadline: projectDeadline },
                success: function (response) {
                    alert('Project saved successfully');
                    location.reload();
                },
                error: function () {
                    alert('Error saving project');
                }
            });
        });

        // Handling Add Candidate Form Submission
        $('#addNewCandidateForm').submit(function (event) {
            event.preventDefault();
            var candidateName = $('#candidateName').val();
            var candidateRole = $('#candidateRole').val();
            var candidatePassword = $('#candidatePassword').val();
            var assignedProject = $('#assignedProject').val();
            $.ajax({
                url: '/candidate/save',
                type: 'POST',
                data: { name: candidateName, role: candidateRole, password: candidatePassword, projectId: assignedProject },
                success: function (response) {
                    alert('Candidate added successfully');
                    location.reload();
                },
                error: function () {
                    alert('Error adding candidate');
                }
            });
        });
    });
</script>

        $("#show_hide_password a").on('click', function (event) {
            event.preventDefault();
            if (passwordInput.attr("type") === "text") {
                passwordInput.attr('type', 'password');
                toggleIcon.addClass("bi-eye-slash-fill").removeClass("bi-eye-fill");
            } else {
                passwordInput.attr('type', 'text');

package organization

import grails.gorm.transactions.Transactional

class UserController {

    def index() {
        // Renders the registration view (register.gsp)
        render(view: "register")
    }

    def dashboard() {
        if (!session.user) {
            // If user is not logged in, redirect to login page
            flash.message = "Please login first"
            redirect(action: "login")
            return
        }
        flash.message = "Welcome, ${session.user.username}"
        render(view: "index")
    }

    def login() {
        // Renders the login view (auth-basic-login.gsp)
        render(view: "auth-basic-login")
    }

    def forgotpassword() {
        // Renders the forgot password view (auth-basic-forgot-password.gsp)
        render(view: "auth-basic-forgot-password")
    }

    def profile() {
        if (!session.user) {
            // If user is not logged in, redirect to login page
            redirect(action: "login")
            return
        }
        // Pass the logged-in user's information to the profile view
        render(view: "user-profile", model: [user: session.user])
    }

    @Transactional
    def save() {
        // Creates a new user with the provided parameters from the registration form
        def user = new User(params)
        // Saves the user to the database
        if (user.save(flush: true)) {
            // If save is successful, display a success message and redirect to the login page
            flash.message = "User registered successfully"
            redirect(action: "login")
        } else {
            // If save fails, display an error message and render the registration view again
            flash.message = "Failed to register user"
            render(view: "register", model: [user: user])
        }
    }

    @Transactional
    def updateProfile() {
        def user = User.get(session.user.id)
        if (user) {
            // Update user profile with the provided parameters
            user.properties = params
            if (user.save(flush: true)) {
                flash.message = "Profile updated successfully"
                session.user = user // Update session user
                redirect(action: "profile")
            } else {
                flash.message = "Failed to update profile"
                render(view: "user-profile", model: [user: user])
            }
        } else {
            flash.message = "User not found"
            redirect(action: "login")
        }
    }

    def authenticate() {
        def user = User.findByEmailAndPassword(params.email, params.password)
        if (user) {
            session.user = user // store user info in session
            flash.message = "Login successful"
            flash.loggedInUser = user.username
            redirect(controller: "user", action: "dashboard")
        } else {
            flash.message = "Invalid email or password. Please enter correct info"
            render(view: "auth-basic-login")
        }
    }
}
